b2MouseJoint* joint;
void bind() {

	b2MouseJointDef jointDef;
	b2BodyDef bodyDef;
	b2Body* tmp = world->GetBodyList();

	jointDef.bodyA = myrect;
	jointDef.bodyB = tmp;
	jointDef.collideConnected = true;
	jointDef.maxForce = 1;
	jointDef.frequencyHz = 5;
	jointDef.dampingRatio = 0.9;
	b2Vec2 v1 = b2Vec2(0, 0);
	b2Vec2 v2 = b2Vec2(0, 0);
	
	joint = (b2MouseJoint*)world->CreateJoint(&jointDef);
	//world->CreateJoint(&jointDef);
	joint->GetAnchorA();
	joint->GetAnchorB();
	if (joint != NULL) {
		b2Vec2 mouseworld = body->GetWorldCenter();
	    joint->SetTarget(mouseworld);
	}
}
